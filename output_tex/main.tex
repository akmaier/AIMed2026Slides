\begin{frame}[t]{Deep learning in medicine}{Known operators}
\vspace*{0.4em}
\begin{center}
\begin{tikzpicture}[
    every node/.style={font=\normalsize},
    transform shape, scale=1.35,
    box/.style ={draw, rounded corners=3pt, minimum height=1.4cm,
                 minimum width=2.6cm, align=center, thick,
                 drop shadow={shadow xshift=0.6mm, shadow yshift=-0.6mm,
                              opacity=0.35}},
    train/.style={box, fill=gray!12},
    known/.style={box, fill=blue!12, draw=blue!60!black},
    arr/.style ={-{Latex[length=3mm]}, thick}
  ]
  \node                                 (in)  {\large $x$};
  \node[train, right=4mm of in]         (t1)  {Trainable\\$\hat f_1$};
  \node[known, right=5mm of t1]         (k1)  {Known op.\\$f_2$};
  \node[train, right=5mm of k1]         (t2)  {Trainable\\$\hat f_3$};
  \node[known, right=5mm of t2]         (k2)  {Known op.\\$f_4$};
  \node[right=4mm of k2]                (out) {\large $\hat f(x)$};
  \draw[arr] (in)  -- (t1);
  \draw[arr] (t1)  -- (k1);
  \draw[arr] (k1)  -- (t2);
  \draw[arr] (t2)  -- (k2);
  \draw[arr] (k2)  -- (out);
\end{tikzpicture}\\[0.5em]
{\small Mix \emph{trainable} layers with deterministic operators
(Fourier transform, gradient descent, projection, \dots).}
\end{center}
\pause
\vspace{0.5em}
\begin{block}{Theorem -- known operators reduce the error bound}
For $f=f_n\circ\dots\circ f_1$ approximated by
$\hat f=\hat f_n\circ\dots\circ \hat f_1$ with per-layer errors
$\varepsilon_i$ and Lipschitz constants $\hat L_j$,
\[
  \varepsilon(\hat f) \;\leq\;
  \sum_{i=1}^{n}\Bigl(\prod_{j=i+1}^{n}\hat L_j\Bigr)\,\varepsilon_i .
\]
Replacing $\hat f_k$ with the \emph{known} $f_k$ forces
$\varepsilon_k=0$ and strictly tightens this bound.
\end{block}
\vspace{-0.3em}
\begin{center}\footnotesize
  Maier et al., \emph{Nature Machine Intelligence}, 2019.
\end{center}
\end{frame}

\begin{frame}[t]{What is agentic AI?}{A simple question}
\vspace*{0.6em}
\begin{flushleft}
  {\Huge\itshape\textquotedblleft How do you drive a car?\textquotedblright}
\end{flushleft}
\vspace*{0.6em}
{\large\color{gray}\itshape Thought for a few seconds\dots}
\vspace*{1.4em}
\begin{itemize}\large
  \item A classic prompt to a large language model.
  \item After a short ``thinking pause'' the answer arrives as
        long-form running text.
  \item No tools, no actions -- the model just \emph{describes} how to drive.
\end{itemize}
\end{frame}

\begin{frame}[t]{What is agentic AI?}{Long text answer of an LLM}
\normalsize
To drive a car, follow these steps:
\begin{itemize}\itemsep=0.3em
  \item \textbf{Get in and adjust:} fasten your \textbf{seat belt}, set the
        \textbf{seat} and \textbf{mirrors} so you can see the road clearly.
  \item \textbf{Start the engine:} press the \textbf{brake pedal}, turn the
        key or push the start button. Make sure the gear selector is in
        \textbf{Park (P)} or \textbf{Neutral (N)}.
  \item \textbf{Engage a gear:} for an automatic, move the lever to
        \textbf{Drive (D)}; for a manual, press the \textbf{clutch} and
        select \textbf{1st gear}.
  \item \textbf{Release the parking brake} and slowly lift your foot off
        the brake. Gently press the \textbf{accelerator} to move forward.
  \item \textbf{Steer} smoothly with both hands and \textbf{check mirrors}
        regularly. Keep a safe distance from the vehicle in front.
  \item \textbf{To stop:} ease off the accelerator, press the brake, and
        bring the car to a complete stop. Shift to \textbf{Park} and apply
        the parking brake before turning off the engine.
\end{itemize}
\vspace{0.6em}
\begin{center}\itshape\color{gray}
Always observe traffic rules and adapt your driving to road and weather.
\end{center}
\vspace{0.4em}
\hfill\textbf{$\Rightarrow$ Lots of words, but no \emph{action}.}\hfill\,
\end{frame}
